```python
def add(x, y):
    return x + y


print(add(1, 2))

f = lambda x, y: x + y

print(f(1, 2))
"""
lambada 表达式
lambda parameter_list:expression
三元表达式
x,y 是否大于y，如果是则返回x 否则返回y
x > y ? X : y
条件为真是返回的结果 if 条件判断  else 条件为假是返回的结果
x if x > y else y
"""
```

Map

```python
list_x = [1, 2, 3, 4]
list_y = [1, 4, 9, 16, 25, 36]


r = map(lambda x, y: x*x + y, list_x, list_y)
print(list(r))
>>[2, 8, 18, 32]

```

Reduce

```python
from functools import reduce

#连续计算，连续调用lambda
list_x = [1, 2, 3, 4]
# list_y = [1, 4, 9, 16, 25, 36]
r = reduce(lambda x, y: x + y, list_x, 20)
print(r)
>>30
```

Filter 过滤

```python
list_x = [1, 0, 1, 0, 0, 1]
list_u = ['a', 'B', 'c', 'F']
r = filter(lambda x: True if x == 1 else False, list_x)
s = filter(lambda u: u.isupper(), list_u)
print(list(r))
print(list(s))
>>[1, 1, 1]
>>['B', 'F']
```

```python
# 装饰器
# 特性 注解
import time


def decorator(func):
    def wrapper():
        print(time.time())
        func()

    return wrapper


def f1():
    # print(int(time.time()))
    print("sssssss")


f = decorator(f1)
f()
>>1599212311.8918478
>> sssssss
```

```python
# 装饰器
# 特性 注解
```
```python
# 使用多个参数
import time


def decorator(func):
    def wrapper(*args):
        print(time.time())
        func(*args)

    return wrapper


@decorator
def f1(func_name):
    # print(int(time.time()))
    print("sssssss" + func_name)


@decorator
def f2(func_name1, func_name2):
    print("qqqqqqq" + func_name1)
    print("qqqqqqq" + func_name2)


# f = decorator(f1)
f1('test')
f2('tt', 'yy')
>>1599213825.521913
>>ssssssstest
>>1599213825.521962
>>qqqqqqqtt
>>qqqqqqqyy
```

```python
# 装饰器
# 特性 注解
import time


def decorator(func):
    def wrapper(*args, **kw):
        print(time.time())
        func(*args, **kw)

    return wrapper


@decorator
def f1(func_name):
    # print(int(time.time()))
    print("sssssss" + func_name)


@decorator
def f2(func_name1, func_name2):
    print("qqqqqqq" + func_name1)
    print("qqqqqqq" + func_name2)


@decorator
def f3(func_name1, func_name2, **kwargs):
    print("qqqqqqq" + func_name1)
    print("qqqqqqq" + func_name2)
    print(kwargs)


# f = decorator(f1)
f1('test')
f2('tt', 'yy')
f3('re', 'ty', a=32, b=56, c='123')


>>1599214333.447851
>>ssssssstest
>>1599214333.4479601
>>qqqqqqqtt
>>qqqqqqqyy
>>1599214333.448002
>>qqqqqqqre
>>qqqqqqqty
```
